## 3. JavaScript逻辑修改

### 3.1 通用函数添加

在每个页面的JavaScript部分添加以下通用函数：

```javascript
// 更新消息红点
function updateMessageBadge(count) {
    var $badge = $('#messageBadge');
    if (count > 0) {
        var displayCount = count > 99 ? '99+' : count;
        $badge.text(displayCount);
        $badge.show();
        $badge.addClass('has-unread');
        
        // 如果数量超过10，添加警告样式
        if (count > 10) {
            $badge.css('background-color', '#ff7875');
        } else {
            $badge.css('background-color', '#ff4d4f');
        }
    } else {
        $badge.hide();
        $badge.removeClass('has-unread');
    }
}

// 获取未读消息数
function fetchUnreadMessageCount(userId) {
    if (!userId) return;
    
    $.ajax({
        url: '/message/unread-count/' + userId,
        type: 'GET',
        dataType: 'json',
        success: function(res) {
            if (res.code === 200) {
                updateMessageBadge(res.data);
            }
        },
        error: function() {
            console.error('获取未读消息数失败');
        }
    });
}

// 定时更新未读消息数（每30秒）
var messagePollingInterval = null;
function startMessagePolling(userId) {
    if (messagePollingInterval) {
        clearInterval(messagePollingInterval);
    }
    
    // 立即获取一次
    fetchUnreadMessageCount(userId);
    
    // 设置定时器
    messagePollingInterval = setInterval(function() {
        fetchUnreadMessageCount(userId);
    }, 30000); // 30秒
}

// 停止轮询
function stopMessagePolling() {
    if (messagePollingInterval) {
        clearInterval(messagePollingInterval);
        messagePollingInterval = null;
    }
}
```

### 3.2 修改现有用户信息获取逻辑

在每个页面的用户信息获取成功后，添加未读消息获取：

```javascript
// 获取当前登录用户信息
$.ajax({
    url: '/user/current',
    type: 'GET',
    dataType: 'json',
    success: function(res) {
        console.log('获取当前用户信息成功:', res);
        if (res.code === 200 && res.data) {
            var user = res.data;
            $('#welcomeText').text('欢迎，' + user.nickname);
            
            // 如果是管理员，显示管理后台菜单
            if (user.role === 1 || user.role === 2) {
                $('.admin-menu').show();
            }
            
            // 获取未读消息数并开始轮询
            fetchUnreadMessageCount(user.id);
            startMessagePolling(user.id);
            
        } else {
            $('#welcomeText').text('欢迎，未知用户');
            console.log('获取当前用户信息失败:', res.message);
        }
    },
    error: function(xhr, status, error) {
        $('#welcomeText').text('欢迎，未知用户');
        console.error('获取当前用户信息请求失败:', error);
    }
});
```

```javascript
// 标记所有消息为已读
window.markAllMessagesAsRead = function() {
    layer.confirm('确定要标记所有消息为已读吗？', {
        title: '确认',
        btn: ['确定', '取消']
    }, function(index) {
        $.ajax({
            url: '/user/current',
            type: 'GET',
            dataType: 'json',
            success: function(res) {
                if (res.code === 200 && res.data) {
                    var userId = res.data.id;
                    $.ajax({
                        url: '/message/read-all/' + userId,
                        type: 'POST',
                        dataType: 'json',
                        success: function(res) {
                            if (res.code === 200) {
                                layer.msg('全部标记为已读成功', {icon: 1});
                                refreshCurrentMessageTab();
                                // 更新红点（设置为0）
                                updateMessageBadge(0);
                            } else {
                                layer.msg('标记失败', {icon: 2});
                            }
                        },
                        error: function() {
                            layer.msg('服务器错误', {icon: 2});
                        }
                    });
                }
            }
        });
        layer.close(index);
    });
};
```
